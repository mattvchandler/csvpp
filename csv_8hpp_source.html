887  {
888  if(state_ != State::consume_newlines)
889  return;

915  std::string parse()
916  {
917  consume_newlines();
918 
919  if(eof())
920  return {};
921 
922  bool quoted = false;
923  std::string field;
924 
925  bool field_done = false;
926  while(!field_done)
927  {
928  int c = getc();
929  bool c_done = false;
930  while(!c_done)
931  {
932  switch(state_)
933  {
934  case State::quote:
935  // end of the field?
936  if(c == delimiter_ || c == '\n' || c == '\r' || c == std::istream::traits_type::eof())
937  {
938  quoted = false;
939  state_ = State::read;
940  break;
941  }
942  // if it's not an escaped quote, then it's an error
943  else if(c == quote_)
944  {
945  field += c;
946  state_ = State::read;
947  c_done = true;
948  break;
949  }
950  else if(lenient_)
951  {
952  field += quote_;
953  field += c;
954  state_ = State::read;
955  c_done = true;
956  break;
957  }
958  else
959  throw Parse_error("Unescaped quote", line_no_, col_no_ - 1);
960 
961  case State::read:
962  // we need special handling for quotes
963  if(c == quote_)
964  {
965  if(quoted)
966  {
967  state_ = State::quote;
968  c_done = true;
969  break;
970  }
971  else
972  {
973  if(field.empty())
974  {
975  quoted = true;
976  c_done = true;
977  break;
978  }
979  else if(!lenient_)
980  {
981  // quotes are not allowed inside of an unquoted field
982  throw Parse_error("quote found in unquoted field", line_no_, col_no_);
983  }
984  }
985  }
986 
987  if(quoted && c == std::istream::traits_type::eof())
988  {
989  if(lenient_)
990  {
991  end_of_row_ = field_done = c_done = true;
992  state_ = State::consume_newlines;
993  break;
994  }
995  else
996  throw Parse_error("Unterminated quoted field - reached end-of-file", line_no_, col_no_);
997  }
998  else if(!quoted && c == delimiter_)
999  {
1000  field_done = c_done = true;
1001  break;
1002  }
1003  else if(!quoted && (c == '\n' || c == '\r' || c == std::istream::traits_type::eof()))
1004  {
1005  end_of_row_ = field_done = c_done = true;
1006  state_ = State::consume_newlines;
1007  break;
1008  }
1009 
1010  field += c;
1011  c_done = true;
1012  break;
1013 
1014  default:
1015  // It should not be possible to reach this state
1016  throw Internal_error{"Illegal state"};
1017  }
1018  }
1019  }
1020  return field;
1021  }
