874  void consume_newlines()
875  {

877  return;

903  std::string parse()
904  {
905  consume_newlines();
906 
907  if(eof())
908  return {};
909 
910  bool quoted = false;
911  std::string field;
912 
913  bool field_done = false;
914  while(!field_done)
915  {
916  int c = getc();
917  bool c_done = false;
918  while(!c_done)
919  {
920  switch(state_)
921  {
922  case State::quote:
923  // end of the field?
924  if(c == delimiter_ || c == '\n' || c == '\r' || c == std::istream::traits_type::eof())
925  {
926  quoted = false;
927  state_ = State::read;
928  break;
929  }
930  // if it's not an escaped quote, then it's an error
931  else if(c == quote_)
932  {
933  field += c;
934  state_ = State::read;
935  c_done = true;
936  break;
937  }
938  else if(lenient_)
939  {
940  field += quote_;
941  field += c;
942  state_ = State::read;
943  c_done = true;
944  break;
945  }
946  else
947  throw Parse_error("Unescaped quote", line_no_, col_no_ - 1);
948 
949  case State::read:
950  // we need special handling for quotes
951  if(c == quote_)
952  {
953  if(quoted)
954  {
955  state_ = State::quote;
956  c_done = true;
957  break;
958  }
959  else
960  {
961  if(field.empty())
962  {
963  quoted = true;
964  c_done = true;
965  break;
966  }
967  else if(!lenient_)
968  {
969  // quotes are not allowed inside of an unquoted field
970  throw Parse_error("quote found in unquoted field", line_no_, col_no_);
971  }
972  }
973  }
974 
975  if(quoted && c == std::istream::traits_type::eof())
976  {
977  if(lenient_)
978  {
979  end_of_row_ = field_done = c_done = true;
980  state_ = State::consume_newlines;
981  break;
982  }
983  else
984  throw Parse_error("Unterminated quoted field - reached end-of-file", line_no_, col_no_);
985  }
986  else if(!quoted && c == delimiter_)
987  {
988  field_done = c_done = true;
989  break;
990  }
991  else if(!quoted && (c == '\n' || c == '\r' || c == std::istream::traits_type::eof()))
992  {
993  end_of_row_ = field_done = c_done = true;
994  state_ = State::consume_newlines;
995  break;
996  }
997 
998  field += c;
999  c_done = true;
1000  break;
1001 
1002  default:
1003  // It should not be possible to reach this state
1004  throw Internal_error{"Illegal state"};
1005  }
1006  }
1007  }
1008  return field;
1009  }
